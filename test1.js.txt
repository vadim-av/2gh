<script>
var wordArr = ['Разнообразный', 'привет', 'мама', 'брат', 'сестра', 'бабушка'];
function renewArray(arr) {
  for (i = 0; i < arr.length; ++i) {
    s = arr[i];
    c = {};
    r = '';
    for (var j = 0; j < s.length; j++) {
      if (!c[s[j]]) {
        r = r + s[j];
        c[s[j]] = 1;
      }
    }
    document.writeln (r);
    arr[i] = r;
  }
}
renewArray(wordArr);
</script>
